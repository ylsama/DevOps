# Directories

```sh
mkdir /tmp/asia
mkdir /tmp/asia/india
mkdir /tmp/asia/india/bangalore


mkdir -p /tmp/asia/india/bangalore
rm -r /tmp/my_dir1

# Copy all content to other Directories
cp -r my_dir1 /tmp/mkdir

```

# File

```sh
# Create new file
touch new_file.txt 

cat > new_file.txt

# View contents of file
cat new_file.txt

# Move/copy/remove {rename} file
mv/cp new_file.txt copy_file.txt 
rm copy_file.txt 
```
